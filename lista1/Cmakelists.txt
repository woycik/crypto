cmake_minimum_required(VERSION 3.14)
project(lista1 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(THREADS_PREFER_PTHREAD_FLAG ON)
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -march=native")
set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -march=native")
set(CMAKE_INTERPROCEDURAL_OPTIMIZATION_RELEASE TRUE)
find_package(Threads REQUIRED)

add_library(zad3_lib STATIC
    zad31.cpp
    zad2.cpp
    md5.c
)

set_source_files_properties(md5.c PROPERTIES LANGUAGE CXX)

target_include_directories(zad3_lib PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_compile_options(zad3_lib PRIVATE -Wall -Wextra -O3)
target_link_libraries(zad3_lib PRIVATE Threads::Threads)

add_executable(md5_attack
    zad3_main.cpp
)
target_link_libraries(md5_attack PRIVATE zad3_lib Threads::Threads)

if(BUILD_TESTING)
    include(FetchContent)
    FetchContent_Declare(
        googletest
        GIT_REPOSITORY https://github.com/google/googletest.git
        GIT_TAG release-1.12.1
    )
    set(gtest_disable_pthreads OFF)
    set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
    FetchContent_MakeAvailable(googletest)

    enable_testing()

    add_executable(runTests
        tests/zad3_test.cpp
    )

    target_link_libraries(runTests PRIVATE zad3_lib gtest_main Threads::Threads)
    target_compile_options(runTests PRIVATE -Wall -Wextra -O2 -pthread)

    include(GoogleTest)
    gtest_discover_tests(runTests)
endif()
